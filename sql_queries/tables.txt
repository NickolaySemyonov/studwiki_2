DROP TABLE wiki_user;
CREATE TABLE wiki_user (
    user_id SERIAL PRIMARY KEY,
    user_email VARCHAR(50) NOT NULL UNIQUE CHECK (user_email ~ '^[a-zA-Z0-9._%+-]+@(gmail\.com|mail\.ru)$'),
    user_pwhash TEXT NOT NULL,
    user_nickname VARCHAR(50) NOT NULL CHECK (user_nickname ~ '^[a-zA-Zа-яА-Я0-9._%+-]{3,50}$'),
    user_role VARCHAR(10) NOT NULL CHECK (user_role IN ('admin', 'moderator', 'user', 'banned'))
);
